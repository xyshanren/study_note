### 第二阶段：模型训练

```python
#多模型集成训练
class MetallurgyProcessRecommender:
	def __init__(self):
		self.rf_model = RandomForestRegressor()
		self.xgb_model = XGBRegressor()
		self.nn_model = MLPRegressor()
	def train(self, X_train, y_train):
		# 并行训练多个模型
		self.rf_model.fit(X_train, y_train)
		self.xgb_model.fit(X_train, y_train)
		self.nn_model.fit(X_train, y_train)
	def predict(self, X_new):
		# 集成预测结果
		rf_pred = self.rf_model.predict(X_new)
		xgb_pred = self.xgb_model.predict(X_new)
		nn_pred = self.nn_model.predict(X_new)
		# 加权平均
		final_pred = 0.4*rf_pred + 0.4*xgb_pred + 0.2*nn_pred return final_pred
```
